Number Range Checker


Write the function checkRange(N) to categorize a given integer N into one of the following ranges:


    

  • Return "negative" if the number is less than 0.
    • 

  • Return "small" if the number is between 0 and 10 (inclusive).
    • 

  • Return "medium" if the number is between 11 and 100 (inclusive).
    • 

  • Return "large" if the number is greater than 100.
    • 



You are provided with a prefilled code structure. Your task is to implement the function checkRange(N).


Input Format


    

  • A single integer N.
    • 



Output Format


    

  • The function should return a string based on the categorization of N:
    • 

  • "negative" if N is less than 0.
    • 

  • "small" if N is between 0 and 10 (inclusive).
    • 

  • "medium" if N is between 11 and 100 (inclusive).
    • 

  • "large" if N is greater than 100.
    • 



Constraints


    

  • -10^3 <= N <= 10^3
    • 



Sample Input:


-5


Sample Output:


negative


Explanation:


    

  • The integer -5 is less than 0, so the function returns "negative".
    • 

"use strict";
process.stdin.resume();
process.stdin.setEncoding("utf-8");
let inputString = "";
let currentLine = 0;
process.stdin.on("data", (inputStdin) => {
    inputString += inputStdin;
});

process.stdin.on("end", (_) => {
    inputString = inputString
        .trim()
        .split("\n")
        .map((str) => str.trim());
    main();

});


//Don't modify above code

function checkRange(N) {
    if(N<0){
        return ("negative");
    }
    else if(N<=10 && N>=0){
        return ("small");
    }
    else if(N>=11 && N<=100){
        return ("medium");
    }
    else{
        return ("large");
    }
}


//Don't modify code below


function readLine() {

    return inputString[currentLine++];

}

function main() {

    let N = parseInt(readLine());
    console.log(checkRange(N));
}

Check the Age Group


Given an age num, categorize the person into the following age groups:


    

  • Return "Child" if the age is less than 13.
    • 

  • Return "Teenager" if the age is between 13 and 19 (inclusive).
    • 

  • Return "Adult" if the age is between 20 and 64 (inclusive).
    • 

  • Return "Senior" if the age is 65 or above.
    • 

  • Return "Invalid Age" if the age is outside the valid range (0 <= num <= 120).
    • 



Input Format


    

  • The input consists of a single integer num representing the age of the person.
    • 



Output Format


    

  • Return a string representing the person's age group based on the specified conditions.
    • 



Sample Input:


16


Sample Output:


Teenager


Explanation:


    

  • The person's age is between 13 and 19, so the output is "Teenager".
    •

function checkAgeGroup(num) {
  if (parseInt(num) >= 0 && parseInt(num) <= 13){
      return "Child";
  }
  else if (parseInt(num) >= 13 && parseInt(num)<=19){
      return "Teenager";
  }
  else if (parseInt(num) >= 20 && parseInt(num)<=64){
      return "Adult";
  }
  else if (parseInt(num) >= 65 && parseInt(num) <=120){
      return "Senior";
  }
  else{
      return "Invalid Age";
  }
 
}

function readLine() {
    return inputString[currentLine++];
}

    let N = parseInt(readLine());
    console.log(checkAgeGroup(N));
